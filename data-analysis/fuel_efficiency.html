from numpy import random
random.seed(0)                                                   # draws a number between 0 and 1 such that all numbers in this interval are equally likely to occur

totals = {20:0, 30:0, 40:0, 50:0, 60:0, 70:0}                    # contains the total number of people in each age group.
purchases = {20:0, 30:0, 40:0, 50:0, 60:0, 70:0}                 # contains the total number of things purchased by people in each age group.
totalPurchases = 0                                               # total of purchases

for _ in range(10_000):                                          # "_" - placeholder, which means that you don't care about the value of the variable, and it is used to loop n times.
    ageDecade = random.choice([20, 30, 40, 50, 60, 70])          # generates 10_000 random "people" and randomly assigns them as being in their 20's, 30's, 40's, 50's, 60's, or 70's.
    purchaseProbability = float(ageDecade) / 100.0
    totals[ageDecade] += 1
    
    if (random.random() < purchaseProbability):                  # random.random() - used to randomly generate floating-point numbers from 0 to 1, that is, 0 <=n <=1.0
        totalPurchases +=1
        purchases[ageDecade] +=1
